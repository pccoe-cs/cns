    let users = {}

    async function hash(p){
      const enc = new TextEncoder().encode(p)
      const buf = await crypto.subtle.digest("SHA-256", enc)
      return [...new Uint8Array(buf)].map(x => x.toString(16).padStart(2,"0")).join("")
    }

    async function register(){
      let u = document.getElementById("regUser").value
      let p = document.getElementById("regPass").value
      if(!u||!p){out("Missing fields", false);return}
      if(users[u]){out("User exists", false);return}
      users[u] = await hash(p)
      out("Registration Successful", true)
    }

    async function login(){
      let u = document.getElementById("logUser").value
      let p = document.getElementById("logPass").value
      if(!u||!p){out("Missing fields", false);return}
      if(!users[u]){out("User not found", false);return}
      let h = await hash(p)
      if(h===users[u]) out("Login Successful", true)
      else out("Invalid Password", false)
    }

    function out(m, success){
      let el = document.getElementById("out")
      el.innerHTML = m
      el.className = success ? "success show" : "error show"
    }
